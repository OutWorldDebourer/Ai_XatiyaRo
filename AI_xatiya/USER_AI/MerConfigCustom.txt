-- Controla si el mercenario perseguirá enemigos en los modos 'Follow' y 'Move'.

MoveChaseDown      = 2    -- [MODO MOVE - PERSECUCIÓN] Activa (1) o desactiva (0) la persecución de enemigos durante el modo 'Move'.

AIDecideChase      = 1    -- [IA - PERSECUCIÓN SELECTIVA] Valor para prevenir que mercenarios más débiles persigan enemigos agresivamente.

-- 6.3. TIEMPOS DE ESPERA Y DISTANCIA MÁXIMA - LIMITES DE OPERACIÓN --

                                -- RECOMENDACIÓN: No exceder 14 celdas para evitar que el mercenario se salga de la pantalla.
